// ===== СОСТОЯНИЕ ===== const state = { user: null, balance: 0, stars: 0, stats: { cases: 0, sold: 0 }, inventory: [] }; // ===== TELEGRAM ===== const tg = window.Telegram.WebApp; tg.expand(); tg.ready(); // универсальная отправка function send(action, data = {}) { tg.sendData(JSON.stringify({ action, ...data })); } // ===== ПРОФИЛЬ ===== function initProfile() { const user = tg.initDataUnsafe?.user; if (!user) return; state.user = user; document.getElementById('userName').textContent = user.first_name + (user.last_name ? ' ' + user.last_name : ''); if (user.photo_url) { document.getElementById('userAvatar').innerHTML = ``; } else { document.getElementById('userAvatar').textContent = user.first_name.charAt(0); } } // ===== ОБНОВЛЕНИЕ UI ===== function updateUI() { document.getElementById('balance').textContent = `❄️ ${state.balance}`; document.getElementById('starsBalance').textContent = `⭐ ${state.stars}`; document.getElementById('statCases').textContent = state.stats.cases; document.getElementById('statSold').textContent = state.stats.sold; updateInventory(); } // ===== КЕЙСЫ ===== function renderCases() { const cases = [ { id: 'trash', name: 'Trash Gifts', price: 750, stars: 75, img: 'https://i.supaimg.com/35109a74-45db-4273-8c8b-a2cf76a8841a/a32b454a-9582-4904-bc0c-ea441b62a589.png' }, { id: 'start', name: 'Start Gifts', price: 750, stars: 75, img: 'https://i.supaimg.com/35109a74-45db-4273-8c8b-a2cf76a8841a/a404bd24-934a-42ad-9db1-d368ecf0e760.png' }, { id: 'frozen', name: 'Frozen Gifts', coming: true } ]; document.getElementById('casesGrid').innerHTML = cases.map(c => `
${c.img ? `` : `
❄️
`}
${c.name}
${ !c.coming ? `
${c.price} ❄️ ${c.stars} ⭐
` : `` }
`).join(''); } function openCase(caseId) { tg.showAlert("❄️ Кейсы пока в разработке"); } // ===== ДЖЕКПОТ ===== function renderJackpot() { const select = document.getElementById('itemSelect'); select.innerHTML = '' + state.inventory.map((item, i) => `` ).join(''); } function placeBet() { const index = document.getElementById('itemSelect').value; if (index === '') return tg.showAlert("❄️ Выбери предмет"); const item = state.inventory[index]; send("place_jackpot_bet", { item_id: item.id }); } // ===== МАГАЗИН ===== function buySnow() { const amount = parseInt(document.getElementById('snowInput').value); if (!amount || amount < 10) return tg.showAlert("❄️ Минимум 10 снежинок"); send("buy", { amount: Math.ceil(amount / 10) * 10 }); } // ===== ИНВЕНТАРЬ ===== let sellingIndex = null; function updateInventory() { const grid = document.getElementById("inventoryGrid"); if (!state.inventory.length) { grid.innerHTML = `
Пусто
`; return; } grid.innerHTML = state.inventory.map((item, i) => `
${item.emoji || "📦"}
${item.name}
${item.value} ❄️
`).join(''); renderJackpot(); } function openSellModal(index) { sellingIndex = index; const item = state.inventory[index]; document.getElementById("sellItemInfo").innerHTML = `Продать ${item.emoji} ${item.name} за ${item.value} ❄️ ?`; document .getElementById("sellModal") .classList.add("active"); } function closeSellModal() { document .getElementById("sellModal") .classList.remove("active"); sellingIndex = null; } function confirmSell() { if (sellingIndex === null) return; const item = state.inventory[sellingIndex]; send("sell_item", { item_id: item.id, price: item.value }); closeSellModal(); } // ===== ТОП ===== function requestTop() { send("get_top"); } function renderTop(users) { const list = document.getElementById("topList"); if (!users || !users.length) { list.innerHTML = `
❄️ Пока нет игроков
`; return; } list.innerHTML = users.map((u, i) => `
${i + 1}
👤
${u.username}
${u.snowflakes} ❄️
`).join(""); } // ===== НАВИГАЦИЯ ===== let currentPage = "cases"; const pages = [ "cases", "jackpot", "shop", "stats", "top" ]; function navigateTo(page) { if (page === currentPage) return; pages.forEach(p => { const el = document.getElementById(p + "Page"); if (!el) return; el.style.display = p === page ? "block" : "none"; }); document .querySelectorAll(".nav-btn") .forEach((btn, i) => { btn.classList.toggle( "active", pages[i] === page ); }); currentPage = page; if (page === "top") requestTop(); if (page === "jackpot") renderJackpot(); } // ===== СООБЩЕНИЯ ОТ БОТА ===== tg.onEvent("message", function(event) { try { const data = JSON.parse(event.data); switch (data.type) { case "balance": state.balance = data.snowflakes || 0; state.stars = data.stars || 0; break; case "inventory": state.inventory = data.items || []; break; case "stats": state.stats.cases = data.casesOpened || 0; state.stats.sold = data.itemsSold || 0; break; case "top": renderTop(data.users); break; case "sell_result": if (data.success && sellingIndex !== null) { state.inventory.splice( sellingIndex, 1 ); sellingIndex = null; } tg.showAlert( data.success ? "✅ Предмет продан!" : "❄️ Ошибка" ); break; case "jackpot_result": tg.showAlert( data.win ? `🎉 Ты выиграл ${data.prize} ❄️!` : "😢 Не повезло..." ); break; } updateUI(); } catch(e) { console.error("FrozenPepe error:", e); } }); // ===== СТАРТ ===== window.onload = function() { initProfile(); renderCases(); // загрузка данных send("get_balance"); send("get_inventory"); send("get_stats"); pages.forEach(p => { const el = document.getElementById(p + "Page"); if (!el) return; if (p !== "cases") el.style.display = "none"; }); };
